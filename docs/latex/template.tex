\newcommand{\makefoccover}{%
  \begin{titlepage}
    \thispagestyle{empty}
    \newgeometry{margin=0pt}

    \begin{tikzpicture}[remember picture, overlay]
      % ── Fondo blanco completo ──
      \fill[white] (current page.south west) rectangle (current page.north east);

      % ── Barra decorativa superior (azul institucional) ──
      \fill[focblue]
        (current page.north west) rectangle
        ([yshift=-2.8cm] current page.north east);

      % ── Marca institucional — logo + texto dentro de la barra azul ──
      $if(logo-path)$
      \foccovernorthwesttext{0.7cm}{-0.10cm}{\includegraphics[height=4.1cm]{$logo-path$}}

      \foccovernortheasttext{-0.9cm}{-0.12cm}{\parbox[c][3.5cm][c]{9.3cm}{%
        \raggedleft\color{white}\bfseries\sffamily\fontsize{15}{15}\selectfont \textbf{\uppercase{$institution-name$}}
      }}
      $else$
      \foccovernortheasttext{-0.9cm}{-0.08cm}{\parbox[c][2.0cm][c]{14cm}{%
        \raggedleft\color{white}\bfseries\sffamily\fontsize{15}{15}\selectfont \textbf{\uppercase{$institution-name$}}
      }}
      $endif$

      % ── Línea divisoria sutil ──
      \draw[focblue, line width=0.4pt]
        ([yshift=-2.8cm] current page.north west) --
        ([yshift=-2.8cm] current page.north east);

      % ── Título "VeriTix" — centrado, parte superior ──
      \foccovercentertext{3.2cm}{\color{focblue}\fontsize{72}{74}\selectfont\bfseries\sffamily $title$}

      % ── Subtítulo — debajo del título ──
      \foccovercentertext{1.2cm}{\color{focmid}\fontsize{17}{20}\selectfont\sffamily $subtitle$}

      % ── Línea horizontal decorativa ──
      \draw[foclight, line width=1pt, opacity=0.85]
        ([xshift=-4.0cm, yshift=-0.4cm] current page.center) --
        ([xshift=4.0cm, yshift=-0.4cm] current page.center);

      % ── Fase del proyecto ──
      \foccovercentertext{-2.2cm}{\color{focblue}\bfseries\sffamily\fontsize{13}{15}\selectfont $if(phase)$$phase$$endif$}
